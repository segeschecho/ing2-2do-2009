\chapter{Planificación de la primera iteración}

\section{División en tareas de los casos de uso para la primera iteración}

En esta sección consideraremos las tareas asociadas a los casos de uso que fueron determinados para la primera iteración por la cátedra. Cabe destacar que sólo se van a considerar un subconjunto de las tareas para cada caso de uso, ya que constituyen una prueba de concepto y la funcionalidad completa se podrá completar en el resto de las iteraciones.

\subsection{Casos de uso asociados a la terminal remota}

\begin{enumerate}
  \item Sincronizar datos de censados (alarma cada 1 minuto).

	\begin{itemize}
			\item Investigación
		\begin{itemize}
			\item Posibles formar de completar datos según métodos numéricos.
			\item Librerias de alarmas y precisión de las mismas(deben tener en cuenta el tiempo REAL que paso y no solo el que tardo la aplicación en ejecutar).
		\end{itemize}
		\item Implementación
			\begin{itemize}
			\item Verificador de datos completos.
			\item Completador de datods según datos históricos.
			\item	Armador de paquete de datos.
			\end{itemize}
		\item Testeo
		 \begin{itemize}
		 \item Armar datos falsos con posibles datos faltantes. 
		 \item Testing de unidad.
		 \end{itemize}
		\item Otros
      \begin{itemize}
      \item Armado de ambiente falso para realizar implementación.
       \item Generación de datos falsos.
			\end{itemize}	
\end{itemize}

\item Encriptar/comprimir datos y enviado según protocolo, al terminar de sincronizar (timestamp).
\begin{itemize}
\item Investigación
\begin{itemize}
\item Protocolo seguro y confiable para la TR con la EC.
\item Modelos de encriptación/compresión de datos (NO para esta iter)
\end{itemize}
\item Implementación
\begin{itemize}
\item Interfaz de capa de encriptación/compresión de datos.
\item Capa de envío de datos segura y confiable.
\end{itemize}
\item Testeo
\item Otros
\begin{itemize}
\item Definir forma de generar datos falsos que luego puedan ser levantados por la EC.
\end{itemize}
\end{itemize}

\item Repetir SMS si no llega el ACK correspondiente.
\begin{itemize}
\item Investigación
\begin{itemize}
	\item Formas de protocolos que permiten reenvío de mensajes.
	\item Cual es el tiempo esperado para el ACK.
\end{itemize}
\item Implementación
\begin{itemize}
	\item Obtención de dato anteriormente enviado sin ACK recibido.
	\item Protocolo de no reenvio dato si llega algun ACK de todos los enviados.
\end{itemize}
\item Testeo
\item Otros
\begin{itemize}
	\item Generación de ACK falsos con probabilidad de arrivo.
\end{itemize}
\end{itemize}

\item Borrar datos que llegaron bien. 
\begin{itemize}
\item Implementación
\begin{itemize}
	\item Buffer de datos historicos ya enviados.
	\item Verificador de existencia de datos y sacarlo.
\end{itemize}
\item Testeo
\item Otros
\begin{itemize}
	\item Generación de datos falsos los cual seran los borrados luego.
\end{itemize}
\end{itemize}
\end{enumerate}

\subsection{Casos de uso asociados a la Estación central}

\begin{enumerate}

\item Procesar modelos matemáticos.
\begin{itemize}
\item Investigación
\begin{itemize}
	\item Modelos matematicos más simples para primera etapa.
\end{itemize}
\item Implementación
\begin{itemize}
	\item Modelos Matemáticos simples.
	\item Simulador de pedido de procesamiento de datos (Ya que no se hace CU1).
\end{itemize}
\item Testeo
\item Otros
\begin{itemize}
	\item Armado de arquitectura que permita procesamiento de modelo.
\end{itemize}
\end{itemize}

\item Detectar caída/alta de TR.
\begin{itemize}
\item Investigación
\begin{itemize}
	\item Motivos de caida de una TR y formas de detección.
\end{itemize}
\item Implementación
\begin{itemize}
	\item Capa de detección de caída por los casos antes mencionados.
	\item Verificador real de que la TR de que la esta caída.
	\item Regenerarador de TR original al detectar alta.
\end{itemize}
\item Testeo
\item Otros
\begin{itemize}
	\item Armado de escenario falsos simulando caída de TR.
\end{itemize}
\end{itemize}

\item Triangulando TR/sensor y anunciarlo.
\begin{itemize}
\item Investigación
\begin{itemize}
	\item Formas de triangular inteligentemnte para minibar error en datos.
\end{itemize}
\item Implementación
\begin{itemize}
	\item Generador de TR virtual.
	\item Simulador de enviador de datos desde la TR virtual.
	\item Generador de datos de dicha TR al tener datos de las TR trianguladas.
\end{itemize}
\item Testeo
\item Otros
\begin{itemize}
	\item Armado de escenario falsos simulando caida de TR.
\end{itemize}
\end{itemize}

\item Ordenar/descartar repetidos de SMS.

\begin{itemize}
\item Investigación
\begin{itemize}
	\item Protocolos de descarte de SMS y formas de darse cuenta de dicho caso.
\end{itemize}
\item Implementación
\begin{itemize}
	\item Almacenamiento de mensajes historicos para verificar si es repetido.
  \item Etiquetador unico para mensaje.
\end{itemize}
\item Testeo
\item Otros
\begin{itemize}
	\item Armado de ambiente para simular mensajes repetidos
\end{itemize}
\end{itemize}

\item Confirmar llegada de SMS con ACK.

\begin{itemize}
\item Investigación
\begin{itemize}
	\item 
Formato de ACK y información extra que pueda ser últil enviar (como número de error).
\end{itemize}
\item Implementación
\begin{itemize}
	\item Recibidor de mensaje.
	\item Verificador de mensaje correcto.
	\item Capa (simple) de desencriptación / descompresión. 
	\item Envío de mensaje con ACK.
\end{itemize}
\item Testeo
\item Otros
\begin{itemize}
	\item Generación de mensajes falsos para poder simular recibirlos y confirmar.
\end{itemize}
\end{itemize}

\item Recibiendo datos de TR.

\begin{itemize}
\item Investigación
\begin{itemize}
	\item Formato de datos a recibir.
\end{itemize}
\item Implementación
\begin{itemize}
	\item Capa de recibo de mensajes (ver CU anterior).
\end{itemize}
\item Testeo
\item Otros
\begin{itemize}
	\item Generar datos falos para poder recibir.
\end{itemize}
\end{itemize}


\end{enumerate}
